Top 7 Ingredients

with cte as (
 select cont.MainContent AS Core_Ingredient,
        count(*) as Ingredient_Count

 from main_contents cont
 inner join country_national_dish cnd
 on cnd.Dishes_ID = cont.Dishes_ID

 group by cont.MainContent
 order by count(*) DESC )

SELECT Core_Ingredient,
       Ingredient_Count,
    ROUND(
[]

               Ingredient_Count                                             * 100, 0) AS Percentage
   SELECT SUM(Ingredient_Count) AS TOTAL FROM cte)


FROM cte
group by Core_Ingredient
ORDER BY Percentage desc
limit 7;
